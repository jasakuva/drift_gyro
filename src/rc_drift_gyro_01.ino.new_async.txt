void IRAM_ATTR onControlTimer() {
  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
  vTaskNotifyGiveFromISR(controlTaskHandle, &xHigherPriorityTaskWoken);
  if (xHigherPriorityTaskWoken) {
    portYIELD_FROM_ISR();
  }
}

static inline int16_t clamp16(int16_t v, int16_t lo, int16_t hi) {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

static inline float lpf(float prev, float x, float alpha) {
  return prev + alpha * (x - prev);
}

static void setupRmtInput(int pin) {
  if (!rmtInit(pin, RMT_RX_MODE, RMT_MEM_NUM_BLOCKS_1, RMT_HZ)) {
    Serial.printf("rmtInit failed on pin %d\n", pin);
    while (1) {
      delay(100);
    }
  }

  // Filter tiny glitches
  rmtSetRxMinThreshold(pin, 50);     // 50 us

  // End frame after idle gap
  rmtSetRxMaxThreshold(pin, 3000);   // 3000 us
}

static bool startRcReadAsync(int pin, size_t &count, size_t capacity, rmt_data_t *buf) {
  count = capacity;
  return rmtReadAsync(pin, buf, &count);
}

static bool parseRcPulseUs(const rmt_data_t *buf, size_t count, uint16_t &pulseUs) {
  for (size_t i = 0; i < count; i++) {
    const rmt_data_t &s = buf[i];

    if (s.level0 == 1 && s.duration0 >= 800 && s.duration0 <= 2200) {
      pulseUs = s.duration0;
      return true;
    }

    if (s.level1 == 1 && s.duration1 >= 800 && s.duration1 <= 2200) {
      pulseUs = s.duration1;
      return true;
    }
  }
  return false;
}

  cp.correction_exp   = prefs_2.getFloat("g_exp", 1.0);
  cp.gyro_lp_hz       = prefs_2.getInt("gyro_lp_hz", 20);
  cp.derivative_lp_hz = prefs_2.getInt("d_lp_hz", 20);

  cp.pid_p            = prefs_2.getFloat("pid_p", 5.0);

  servoin_lp.setCutoff(cp.steer_in_lp_hz);

static void calibrateGyroOffset() {
  const int samples = 500;
  float sum = 0.0f;

  sensors_event_t a, g, t;

  Serial.println("Calibrating MPU6050 gyro offset... keep car still");
  delay(500);

  for (int i = 0; i < samples; i++) {
    mpu.getEvent(&a, &g, &t);
    sum += g.gyro.z * 57.2957795f;  // rad/s -> deg/s
    delay(2);
  }

  gyroZOffset_dps = sum / samples;
  Serial.print("gyroZOffset_dps=");
  Serial.println(gyroZOffset_dps);
}

void reloadSettings() {
  gyro_lp.setCutoff(cp.gyro_lp_hz);
  derivative_lp.setCutoff(cp.derivative_lp_hz);
  servoin_lp.setCutoff(cp.steer_in_lp_hz);
  servoout_lp.setCutoff(cp.steer_out_lp_hz);
  wob.setAmplitude(cp.wobble_det_a);
  driftd.init(LOOP_PERIOD_S, cp.dd_min_steer, cp.dd_min_yaw);
}

void rcInputTask(void* pvParameters) {
  uint16_t newPulse = 0;

  // Arm async reads once at task start
  if (!steerReadArmed) {
    steerReadArmed = startRcReadAsync(
      PIN_STEER_IN,
      steerCount,
      RMT_SYMBOLS_OF(steerBuf),
      steerBuf
    );
  }

  if (!gainReadArmed) {
    gainReadArmed = startRcReadAsync(
      PIN_GAIN_IN,
      gainCount,
      RMT_SYMBOLS_OF(gainBuf),
      gainBuf
    );
  }

  for (;;) {
    // STEER
    if (steerReadArmed && rmtReceiveCompleted(PIN_STEER_IN)) {
      if (parseRcPulseUs(steerBuf, steerCount, newPulse)) {
        portENTER_CRITICAL(&s_mux);
        s_pw = newPulse;
        s_last_valid_us = micros();
        portEXIT_CRITICAL(&s_mux);
      }

      steerReadArmed = startRcReadAsync(
        PIN_STEER_IN,
        steerCount,
        RMT_SYMBOLS_OF(steerBuf),
        steerBuf
      );
    } else if (!steerReadArmed) {
      steerReadArmed = startRcReadAsync(
        PIN_STEER_IN,
        steerCount,
        RMT_SYMBOLS_OF(steerBuf),
        steerBuf
      );
    }

    // GAIN
    if (gainReadArmed && rmtReceiveCompleted(PIN_GAIN_IN)) {
      if (parseRcPulseUs(gainBuf, gainCount, newPulse)) {
        portENTER_CRITICAL(&s_mux);
        g_pw = newPulse;
        g_last_valid_us = micros();
        portEXIT_CRITICAL(&s_mux);
      }

      gainReadArmed = startRcReadAsync(
        PIN_GAIN_IN,
        gainCount,
        RMT_SYMBOLS_OF(gainBuf),
        gainBuf
      );
    } else if (!gainReadArmed) {
      gainReadArmed = startRcReadAsync(
        PIN_GAIN_IN,
        gainCount,
        RMT_SYMBOLS_OF(gainBuf),
        gainBuf
      );
    }

    vTaskDelay(pdMS_TO_TICKS(1));
  }
}

void wifiTask(void* pvParameters) {
  setupSettings();

  while (makeSettings()) {
    vTaskDelay(pdMS_TO_TICKS(50));
  }

  wifiTaskHandle = nullptr;
  vTaskDelete(nullptr);
}

void controlTask(void* pvParameters) {
  uint32_t lastTickUs = micros();

  for (;;) {
    ulTaskNotifyTake(pdTRUE, portMAX_DELAY);

    if (settings_changed) {
      reloadSettings();
      settings_changed = false;
    }

    uint32_t nowUs = micros();
    float dt_s_local = (nowUs - lastTickUs) * 1e-6f;
    lastTickUs = nowUs;
    (void)dt_s_local;

    ControlParams cp_local;
    portENTER_CRITICAL(&cp_mux);
    cp_local = cp;
    portEXIT_CRITICAL(&cp_mux);

    // Read latest RC pulse widths atomically
    uint16_t steerIn, gainIn;
    uint32_t steerAgeUs, gainAgeUs;

    portENTER_CRITICAL(&s_mux);
    steerIn = s_pw;
    gainIn  = g_pw;
    steerAgeUs = micros() - s_last_valid_us;
    gainAgeUs  = micros() - g_last_valid_us;
    portEXIT_CRITICAL(&s_mux);

    // Simple failsafe
    if (steerAgeUs > 100000) steerIn = RC_MID;   // 100 ms
    if (gainAgeUs  > 100000) gainIn  = RC_MID;

    float gain = ((float)gainIn - 1000.0f) / 1000.0f;
    gain = clamp16((int16_t)(gain * 1000), 0, 1000) / 1000.0f;

    steerIn = servoin_lp.update(steerIn);
    normSteering = mySteeringMap.getNormalized(steerIn);

    // Read MPU6050
    sensors_event_t a, g, t;
    mpu.getEvent(&a, &g, &t);

    gyrodps = (g.gyro.z * 57.2957795f) - gyroZOffset_dps;
    Yaw = gyrodps;

    yawRateFilt = gyro_lp.update(gyrodps);
    float yawDerivativeRaw = dYawRate.update(gyrodps, LOOP_PERIOD_MS);
    filtered_yaw_derivative = derivative_lp.update(yawDerivativeRaw);

    drift_value = driftd.update(normSteering, yawRateFilt, cp.gain);
    float drift_multiplier = 1.0f + (drift_value / 10) * cp.dd_multiplier;

    gyro_correction = drift_multiplier * cp.gain * gain *
                      (cp.pid_p * yawRateFilt + cp.pid_d * filtered_yaw_derivative);

    corr = gyro_correction /
           (1.0f + (abs(dSteering.update(normSteering)) / 2.0f) * cp.steering_prio);

    if (corr != 0.0f) {
      corr = (corr > 0.0f ? 1.0f : -1.0f) * powf(fabs(corr), cp.correction_exp);
    } else {
      corr = 0.0f;
    }

    float delta = corr - lastGyroCorrection;
    if (fabs(delta) > cp.max_d_corr) {
      corr = lastGyroCorrection + (delta > 0 ? cp.max_d_corr : -cp.max_d_corr);
    }

    lastGyroCorrection = corr;

    int16_t out = steerIn + (int16_t)corr;
    out = clamp16(out, epa_low_us_2, epa_high_us_2);

    out = servoout_lp.update(out);
    steerServo.writeMicroseconds(out);

    static uint32_t lastPrint = 0;
    if ((millis() - lastPrint > 200) &&
        (steerAgeUs < 1000000) &&
        cp.debug_serial == 1) {
      lastPrint = millis();
      Serial.print("corr="); Serial.print(corr);
      Serial.print(" Result_gyro="); Serial.print(gyrodps);
      Serial.print(" gyro_correction="); Serial.print(gyro_correction);
      Serial.print(" yawRateFilt="); Serial.print(yawRateFilt);
      Serial.print(" normSteering="); Serial.print(normSteering);
      Serial.print(" gain="); Serial.print(gain);
      Serial.print(" steerIn="); Serial.print(steerIn);
      Serial.print(" out="); Serial.print(out);
      Serial.print(" drift_value="); Serial.print(drift_value);
      Serial.print(" filtered_yaw_derivative="); Serial.println(filtered_yaw_derivative);
    }
  }
}

// -----------------------------------------------------------------------------
// Setup / Loop
// -----------------------------------------------------------------------------
void setup() {
  Serial.begin(115200);

  loadSettings_2();

  Wire.begin(SDA_PIN, SCL_PIN);
  Wire.setClock(400000);
  delay(50);

  if (!mpu.begin(0x68, &Wire)) {
    Serial.println("MPU6050 not found");
    while (1) {
      delay(10);
    }
  }

  Serial.println("MPU6050 found");

  mpu.setAccelerometerRange(MPU6050_RANGE_8_G);
  mpu.setGyroRange(MPU6050_RANGE_1000_DEG);
  mpu.setFilterBandwidth(MPU6050_BAND_94_HZ);

  calibrateGyroOffset();

  pinMode(PIN_STEER_IN, INPUT);
  pinMode(PIN_GAIN_IN, INPUT);
  delay(200);

  setupRmtInput(PIN_STEER_IN);
  setupRmtInput(PIN_GAIN_IN);

  // Optional: arm once here too
  steerReadArmed = startRcReadAsync(
    PIN_STEER_IN,
    steerCount,
    RMT_SYMBOLS_OF(steerBuf),
    steerBuf
  );

  gainReadArmed = startRcReadAsync(
    PIN_GAIN_IN,
    gainCount,
    RMT_SYMBOLS_OF(gainBuf),
    gainBuf
  );

  steerServo.attach(PIN_SERVO_OUT);
  steerServo.setTimerWidth(14);
  steerServo.writeMicroseconds(RC_MID);

  delay(200);

  if (s_pw < 1300 || s_pw > 1700) {
    makeSettings();
  }

  loadSettings_2();
  delay(300);

  // RC input task on same core as control
  xTaskCreatePinnedToCore(
    rcInputTask,
    "rcInputTask",
    4096,
    nullptr,
    3,
    &rcInputTaskHandle,
    1
  );

  // Control task on core 1
  xTaskCreatePinnedToCore(
    controlTask,
    "controlTask",
    8192,
    nullptr,
    4,
    &controlTaskHandle,
    1
  );

  // WiFi/settings task on core 0
  xTaskCreatePinnedToCore(
    wifiTask,
    "wifiTask",
    8192,
    nullptr,
    2,
    &wifiTaskHandle,
    0
  );

  controlTimer = timerBegin(1000000);   // 1 MHz
  timerAttachInterrupt(controlTimer, &onControlTimer);
  timerAlarm(controlTimer, LOOP_PERIOD_US, true, 0);
}

void loop() {
  vTaskDelay(pdMS_TO_TICKS(1000));
}
